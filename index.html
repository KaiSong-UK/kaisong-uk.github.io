# etl-toolkit — define, observe, ship pipelines from etl_toolkit import Pipeline, ReadDatabase, QualityCheck, Transform, WriteDatabase pipeline = (Pipeline("daily_user_sync") .add_step(ReadDatabase( connection="postgresql://user:pass@host/db", query="SELECT * FROM users WHERE updated_at > :last_run" )) .add_step(QualityCheck( name="check_completeness", rules=[{ "type": "not_null", "columns": ["user_id", "email"] }], on_fail="warn" )) .add_step(Transform( name="normalize_email", fn=lambda rows: [{**r, "email": r["email"].lower()} for r in rows] )) .add_step(WriteDatabase( connection="postgresql://...", table="dim_users", mode="upsert", key_columns=["user_id"] )) ) result = pipeline.run() print(f"Done. {result.rows_written} rows in {result.total_duration_ms:.1f}ms")
